`timescale 1ns/1ps

module frame_config_cdc #(
    parameter int ADDR_W = 32
) (
    input  logic               src_clk,
    input  logic               src_rst,
    input  logic               cfg_valid_src,
    input  logic [ADDR_W-1:0]  src_base_addr_src,
    input  logic [ADDR_W-1:0]  dst_base_addr_src,
    input  logic [ADDR_W-1:0]  src_stride_src,
    input  logic [ADDR_W-1:0]  dst_stride_src,
    input  logic [15:0]        src_w_src,
    input  logic [15:0]        src_h_src,
    input  logic [15:0]        dst_w_src,
    input  logic [15:0]        dst_h_src,
    input  logic signed [31:0] rot_sin_q16_src,
    input  logic signed [31:0] rot_cos_q16_src,
    input  logic               cache_prefetch_en_src,
    input  logic [15:0]        scheduler_lead_pixels_src,
    input  logic [7:0]         scheduler_merge_max_x_eff_src,
    input  logic [7:0]         scheduler_merge_min_x_src,
    input  logic [15:0]        scheduler_fifo_depth_eff_src,
    input  logic [15:0]        scheduler_fifo_age_limit_src,
    input  logic [15:0]        scheduler_throttle_cycles_src,
    input  logic [1:0]         scheduler_policy_src,
    output logic               cfg_ready_src,

    input  logic               dst_clk,
    input  logic               dst_rst,
    output logic               cfg_valid_dst,
    output logic [ADDR_W-1:0]  src_base_addr_dst,
    output logic [ADDR_W-1:0]  dst_base_addr_dst,
    output logic [ADDR_W-1:0]  src_stride_dst,
    output logic [ADDR_W-1:0]  dst_stride_dst,
    output logic [15:0]        src_w_dst,
    output logic [15:0]        src_h_dst,
    output logic [15:0]        dst_w_dst,
    output logic [15:0]        dst_h_dst,
    output logic signed [31:0] rot_sin_q16_dst,
    output logic signed [31:0] rot_cos_q16_dst,
    output logic               cache_prefetch_en_dst,
    output logic [15:0]        scheduler_lead_pixels_dst,
    output logic [7:0]         scheduler_merge_max_x_eff_dst,
    output logic [7:0]         scheduler_merge_min_x_dst,
    output logic [15:0]        scheduler_fifo_depth_eff_dst,
    output logic [15:0]        scheduler_fifo_age_limit_dst,
    output logic [15:0]        scheduler_throttle_cycles_dst,
    output logic [1:0]         scheduler_policy_dst,
    input  logic               cfg_ready_dst
);

    typedef struct packed {
        logic [1:0]         scheduler_policy;
        logic [15:0]        scheduler_throttle_cycles;
        logic [15:0]        scheduler_fifo_age_limit;
        logic [15:0]        scheduler_fifo_depth_eff;
        logic [7:0]         scheduler_merge_min_x;
        logic [7:0]         scheduler_merge_max_x_eff;
        logic [15:0]        scheduler_lead_pixels;
        logic               cache_prefetch_en;
        logic signed [31:0] rot_cos_q16;
        logic signed [31:0] rot_sin_q16;
        logic [15:0]        dst_h;
        logic [15:0]        dst_w;
        logic [15:0]        src_h;
        logic [15:0]        src_w;
        logic [ADDR_W-1:0]  dst_stride;
        logic [ADDR_W-1:0]  src_stride;
        logic [ADDR_W-1:0]  dst_base_addr;
        logic [ADDR_W-1:0]  src_base_addr;
    } cfg_payload_t;

    localparam int PAYLOAD_W = $bits(cfg_payload_t);

    cfg_payload_t fifo_wr_payload;
    cfg_payload_t fifo_rd_payload;
    cfg_payload_t dst_payload_reg;
    logic [PAYLOAD_W-1:0] fifo_wr_data;
    logic [PAYLOAD_W-1:0] fifo_rd_data;
    logic dst_valid_reg;
    logic fifo_full;
    logic fifo_empty;
    logic fifo_almost_full;
    logic fifo_overflow;
    logic fifo_underflow;
    logic [3:0] fifo_wr_count;
    logic [3:0] fifo_rd_count;

    always_comb begin
        fifo_wr_payload.src_base_addr = src_base_addr_src;
        fifo_wr_payload.dst_base_addr = dst_base_addr_src;
        fifo_wr_payload.src_stride = src_stride_src;
        fifo_wr_payload.dst_stride = dst_stride_src;
        fifo_wr_payload.src_w = src_w_src;
        fifo_wr_payload.src_h = src_h_src;
        fifo_wr_payload.dst_w = dst_w_src;
        fifo_wr_payload.dst_h = dst_h_src;
        fifo_wr_payload.rot_sin_q16 = rot_sin_q16_src;
        fifo_wr_payload.rot_cos_q16 = rot_cos_q16_src;
        fifo_wr_payload.cache_prefetch_en = cache_prefetch_en_src;
        fifo_wr_payload.scheduler_lead_pixels = scheduler_lead_pixels_src;
        fifo_wr_payload.scheduler_merge_max_x_eff = scheduler_merge_max_x_eff_src;
        fifo_wr_payload.scheduler_merge_min_x = scheduler_merge_min_x_src;
        fifo_wr_payload.scheduler_fifo_depth_eff = scheduler_fifo_depth_eff_src;
        fifo_wr_payload.scheduler_fifo_age_limit = scheduler_fifo_age_limit_src;
        fifo_wr_payload.scheduler_throttle_cycles = scheduler_throttle_cycles_src;
        fifo_wr_payload.scheduler_policy = scheduler_policy_src;
    end

    assign fifo_wr_data = fifo_wr_payload;
    assign fifo_rd_payload = fifo_rd_data;

    assign cfg_ready_src = !fifo_full && (fifo_wr_count == '0);
    assign cfg_valid_dst = dst_valid_reg;
    assign src_base_addr_dst = dst_payload_reg.src_base_addr;
    assign dst_base_addr_dst = dst_payload_reg.dst_base_addr;
    assign src_stride_dst = dst_payload_reg.src_stride;
    assign dst_stride_dst = dst_payload_reg.dst_stride;
    assign src_w_dst = dst_payload_reg.src_w;
    assign src_h_dst = dst_payload_reg.src_h;
    assign dst_w_dst = dst_payload_reg.dst_w;
    assign dst_h_dst = dst_payload_reg.dst_h;
    assign rot_sin_q16_dst = dst_payload_reg.rot_sin_q16;
    assign rot_cos_q16_dst = dst_payload_reg.rot_cos_q16;
    assign cache_prefetch_en_dst = dst_payload_reg.cache_prefetch_en;
    assign scheduler_lead_pixels_dst = dst_payload_reg.scheduler_lead_pixels;
    assign scheduler_merge_max_x_eff_dst = dst_payload_reg.scheduler_merge_max_x_eff;
    assign scheduler_merge_min_x_dst = dst_payload_reg.scheduler_merge_min_x;
    assign scheduler_fifo_depth_eff_dst = dst_payload_reg.scheduler_fifo_depth_eff;
    assign scheduler_fifo_age_limit_dst = dst_payload_reg.scheduler_fifo_age_limit;
    assign scheduler_throttle_cycles_dst = dst_payload_reg.scheduler_throttle_cycles;
    assign scheduler_policy_dst = dst_payload_reg.scheduler_policy;

    async_word_fifo #(
        .DATA_W(PAYLOAD_W),
        .DEPTH(16),
        .ALMOST_FULL_MARGIN(2)
    ) u_cfg_fifo (
        .wr_clk(src_clk),
        .wr_rst(src_rst),
        .wr_en(cfg_valid_src && cfg_ready_src),
        .wr_data(fifo_wr_data),
        .full(fifo_full),
        .almost_full(fifo_almost_full),
        .wr_count(fifo_wr_count),
        .overflow(fifo_overflow),

        .rd_clk(dst_clk),
        .rd_rst(dst_rst),
        .rd_en(!dst_valid_reg && !fifo_empty),
        .rd_data(fifo_rd_data),
        .empty(fifo_empty),
        .rd_count(fifo_rd_count),
        .underflow(fifo_underflow)
    );

    always_ff @(posedge dst_clk) begin
        if (dst_rst) begin
            dst_valid_reg <= 1'b0;
            dst_payload_reg <= '0;
        end else begin
            if (!dst_valid_reg && !fifo_empty) begin
                dst_payload_reg <= fifo_rd_payload;
                dst_valid_reg <= 1'b1;
            end else if (dst_valid_reg && cfg_ready_dst) begin
                dst_valid_reg <= 1'b0;
            end
        end
    end

`ifndef SYNTHESIS
    always_ff @(posedge src_clk) begin
        if (!src_rst && cfg_valid_src && !cfg_ready_src) begin
            $error("frame_config_cdc overflow: cfg_valid_src asserted while FIFO full");
        end
    end
`endif

endmodule
